# (1) formating variables
df <- df %>%
  mutate(
    site = factor(site, levels = c("CRC", "PTR", "OWC", "MSM")),       
    datetime = force_tz(ymd_hms(datetime), tz = "EST"),             
    source = as.factor(source),                                        
    date = as_date(datetime),                                         
    month = as.factor(format(datetime, "%b")),                         
    year = as.factor(year(datetime)),                                  
    time_day_c = format(datetime, "%H:%M"),
    time_day = as.POSIXct(time_day_c, format = "%H:%M", tz = "EST"),
    Eh = orp_mv - (0.718 * temp_c) + 224.41) %>%
  dplyr::select(site, source, datetime, time_day, time_day_c, date, month, year, everything())

# (3) Calculating diel averages and standard deviations for fig 1
summary_df <- df %>%
  filter(flag_out_range_orp == FALSE,
         flag_out_range_ph == FALSE,
         flag_out_of_water == FALSE,
         flag_out_of_SPC == FALSE,
         flag_out_DO_range == FALSE,
         flag_PAR_out == FALSE) %>%
  group_by(site, 
           time_day
  ) %>%
  summarise(
    n_obs_Eh = sum(!is.na(Eh)),
    mean_Eh = mean(Eh, na.rm = TRUE),
    sd_Eh = sd(Eh, na.rm = TRUE),
    min_Eh = min(Eh, na.rm = TRUE),
    
    mean_DO = mean(odo_mg_l, na.rm = TRUE),
    sd_DO = sd(odo_mg_l, na.rm = TRUE),
    
    mean_ph = mean(ph, na.rm = TRUE),
    sd_ph = sd(ph, na.rm = TRUE),
    
    mean_par = mean(par_den_c_avg, na.rm = TRUE),
    sd_par = sd(par_den_c_avg, na.rm = TRUE),
    
    mean_T = mean(temp_c, na.rm = TRUE),
    sd_T = sd(temp_c, na.rm = TRUE)
  )

slope_robusto <- df %>%
  filter(
    flag_out_range_orp == FALSE,
    flag_out_range_ph == FALSE,
    flag_out_of_water == FALSE,

    flag_out_DO_range == FALSE,

  ) %>%
  group_by(site, date) %>%
  filter(n() > 10) %>% 
  summarise(
    slope_rlm = if (n() >= 10) {
      coef(robustbase::lmrob(Eh ~ odo_mg_l))[2]
    } else {
      NA_real_
    },
    n = n(),
    .groups = "drop"
  ) %>%
  filter(!is.na(slope_rlm))

# (2) Eh vs pH --------------------------------------------------------
slope_robusto2 <- df %>%
  filter(
    flag_out_range_orp == FALSE,
    flag_out_range_ph == FALSE,
    flag_out_of_water == FALSE,
    flag_out_of_SPC == FALSE,
    flag_out_DO_range == FALSE,
    odo_mg_l >= 0
  ) %>%
  group_by(site, date) %>%
  filter(n() > 10) %>%
  summarise(
    result = list(
      tryCatch({
        fit <- robustbase::lmrob(Eh ~ ph)
        slope <- coef(fit)[2]
        pval  <- summary(fit)$coefficients[2, 4]
        tibble(slope = slope, p_value = pval)
      }, error = function(e) tibble(slope = NA_real_, p_value = NA_real_),
      warning = function(w) tibble(slope = NA_real_, p_value = NA_real_)
      )
    ),
    n = n(),
    .groups = "drop"
  ) %>%
  tidyr::unnest(result) %>%
  filter(!is.na(p_value), p_value < 0.05)
